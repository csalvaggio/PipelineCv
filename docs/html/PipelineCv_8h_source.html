4 /**
5  * @file PipelineCV.h
6  * @brief Functional-style image processing utilities built on OpenCV.
7  *
8  * This header defines a minimal "pipe" operator (`|`) and a collection of
9  * composable adapters returning `cv::Mat(const cv::Mat&)` callables.
10  * They can be chained together for readable, declarative pipelines:
11  *
12  * @code
13  * cv::Mat img = cv::imread("foo.png");
14  * cv::Mat result = img
15  * | pcv::to_float() // convert to 32F in [0,1]
16  * | pcv::to_gray() // grayscale
17  * | pcv::normalize(0.0, 1.0) // normalize min->0, max->1
18  * | pcv::multiply_scalar(2.0) // scale
19  * | pcv::clamp(0.0, 1.0); // clip
20  * @endcode
21  *
22  * @note All functions are defined `inline` for header-only inclusion.
23  * @see https://docs.opencv.org/
24  *
25  * @author Carl Salvaggio
26  * @date 2025-10-06
27  * @version 1.0
28  * @license MIT
29  */

536 /**
537  * @brief Complex division of two 2-channel images with epsilon stabilization.
538  *
539  * Computes `(a+ib) / (c+id)` using `(ac+bd)/(c^2+d^2+epsilon)` for the real
540  * part and `(bc-ad)/(c^2+d^2+epsilon)` for the imaginary part of the quotient.
541  *
542  * @param other Complex denominator with 2 channels.
543  * @param epsilon Small value added to denominator to avoid division by zero.
544  * @return Callable producing complex quotient.
545  *
546  * @pre Both inputs must have 2 channels.
547  *
548  * \ingroup complex_domain_operations
549  */
550 inline auto complex_divide(const cv::Mat& other, double epsilon = 1e-12) {
551  return [=](const cv::Mat& in) {
552  CV_Assert(in.channels() == 2 && other.channels() == 2);
553  std::vector<cv::Mat> a(2), b(2), result(2);
554  cv::split(in, a);
555  cv::split(other, b);
556  cv::Mat denominator = b[0].mul(b[0]) + b[1].mul(b[1]) + epsilon;
557  result[0] = (a[0].mul(b[0]) + a[1].mul(b[1])) / denominator; // real
558  result[1] = (a[1].mul(b[0]) - a[0].mul(b[1])) / denominator; // imag
559  cv::Mat out;
560  cv::merge(result, out);
561  return out;
562  };

627 /**
628  * @brief Swap DFT quadrants so that the zero-frequency component moves to
629  * the center.
630  *
631  * Works on 2-dimensional cv::Mats (even dimensions recommended).
632  *
633  * @return Callable performing quadrant swaps (Q0<->Q3 and Q1<->Q2).
634  *
635  * \ingroup fourier_frequency_domain_utilities
636  */
637 inline auto fft_shift() {
638  return [](const cv::Mat& in) {
639  cv::Mat out = in.clone();
640  int cx = out.cols / 2;
641  int cy = out.rows / 2;
642  cv::Mat q0(out, cv::Rect(0, 0, cx, cy));
643  cv::Mat q1(out, cv::Rect(cx, 0, cx, cy));
644  cv::Mat q2(out, cv::Rect(0, cy, cx, cy));
645  cv::Mat q3(out, cv::Rect(cx, cy, cx, cy));
646  cv::Mat tmp;
647  q0.copyTo(tmp);
648  q3.copyTo(q0);
649  tmp.copyTo(q3);
650  q1.copyTo(tmp);
651  q2.copyTo(q1);
652  tmp.copyTo(q2);
653  return out;

661 /**
662  * @brief Circularly shift (roll) rows and columns.
663  *
664  * Positive shifts move content down/right; negative move content up/left.
665  *
666  * @param shift_rows Row shift (can be negative).
667  * @param shift_cols Column shift (can be negative).
668  * @return Callable producing a rolled/shifted image.
669  *
670  * @pre `in.dims == 2`.
671  *
672  * \ingroup spatial_utilities
673  */
674 inline auto roll(int shift_rows, int shift_cols) {
675  return [=](const cv::Mat& in) {
676  CV_Assert(in.dims == 2);
677 
678  const int rows = in.rows;
679  const int cols = in.cols;
680 
681  int dy = ((shift_rows % rows) + rows) % rows;
682  int dx = ((shift_cols % cols) + cols) % cols;
683 
684  if (dy == 0 && dx == 0) return in;
685 
686  cv::Mat out(in.size(), in.type());
687

710  * @param scale Uniform scale factor (1.0 = no scaling).
711  * @param tx Horizontal translation in pixels (right positive).
712  * @param ty Vertical translation in pixels (down positive).
713  * @param dsize Output size; (0,0) => same as input.
714  * @param flags Interpolation/warp flags (e.g. cv::INTER_LINEAR).
715  * @param borderMode Border handling (e.g. cv::BORDER_CONSTANT).
716  * @param borderValue Border value used with CONSTANT border mode.
717  * @return Callable applying the specified affine transform.
718  *
719  * \ingroup spatial_utilities
720  */
721 inline auto warp_affine(double angle_degrees, double scale, double tx,
722  double ty, cv::Size dsize = cv::Size(),
723  int flags = cv::INTER_LINEAR,
724  int borderMode = cv::BORDER_CONSTANT,
725  const cv::Scalar& borderValue = cv::Scalar()) {
726  return [=](const cv::Mat& in) {

743  * @brief Polar (or log-polar) warp from Cartesian input using cv::warpPolar.
744  *
745  * This maps a Cartesian image to its polar representation. The x dimension
746  * of the output parameterizes angle in [0, 2Ï€), and the y dimension
747  * parameterizes radius in [0, maxRadius].
748  *
749  * @param log_polar If true, use logarithmic radius mapping; otherwise linear.
750  * @param center Polar center; if either component is negative, uses
751  * image center.
752  * @param maxRadius Maximum radius in pixels; if <= 0, uses farthest corner.
753  * @param dsize Output size; if (0,0), uses input size (same rows/cols).
754  * @param flags Interpolation flags (e.g., cv::INTER_LINEAR).
755  * @return Callable producing the polar/log-polar image.
756  *
757  * \ingroup spatial_utilities
758  */
759 inline auto warp_polar(bool log_polar = false,
760  cv::Point2f center = {-1.f, -1.f},
761  double maxRadius = -1.0, cv::Size dsize = cv::Size(),
762  int flags = cv::INTER_LINEAR) {
763  return [=](const cv::Mat& in) {
764  // Resolve output sampling (Angle x Radius)
765  const cv::Size outSize =
766  (dsize.width > 0 && dsize.height > 0) ? dsize : in.size();
767  // Resolve center (Default: image center)
768  cv::Point2f c = center;
769  if (c.x < 0.f || c.y < 0.f) {
770  c = cv::Point2f(static_cast<float>(in.cols) * 0.5f,
771  static_cast<float>(in.rows) * 0.5f);
772  }
773  // Resolve max radius (default: farthest corner to cover whole image)
774  double mr = maxRadius;
775  if (mr <= 0.0) {
776  const cv::Point2f tl(0.f, 0.f), tr(static_cast<float>(in.cols), 0.f);
777  const cv::Point2f bl(0.f, static_cast<float>(in.rows));
778  const cv::Point2f br(static_cast<float>(in.cols),
779  static_cast<float>(in.rows));
780  auto dist = [&](const cv::Point2f& p) { return cv::norm(p - c); };
781  mr = std::max({dist(tl), dist(tr), dist(bl), dist(br)});
782  }
783  // Mode: linear or logarithmic radial mapping; no inverse mapping exposed
784  const int mode = (log_polar ? cv::WARP_POLAR_LOG : cv::WARP_POLAR_LINEAR);
785  cv::Mat out;
786  out = cv::Mat::zeros(outSize, in.type());
787  cv::warpPolar(in, out, outSize, c, mr, flags | mode);
788  return out;
789  };
